505 std::vector<bool> to_vector() const
506 {
507 uint64_t _size = this->size();
508 std::vector<bool> r;
509 r.resize(_size, false);
510 uint64_t counter = 0;
511 auto _end = this->get_leaf_forward_iterator_end();
512 for (BitForwardIterator it = this->get_bit_forward_iterator_begin(); it != _end; ++it)
513 {
514 uint64_t bits = *it;
515 uint64_t i = 0;
516 while (i < 64 && counter < _size)
517 {
518 bool b = stool::Byte::get_bit(bits, i);
519 r[counter] = b;
520 counter++;
521 i++;
522 }
523 }
524 assert(counter == _size);
