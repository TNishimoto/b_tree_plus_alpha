504 std::vector<bool> to_vector() const
505 {
506 uint64_t _size = this->size();
507 std::vector<bool> r;
508 r.resize(_size, false);
509 uint64_t counter = 0;
510 auto _end = this->get_leaf_forward_iterator_end();
511 for (BitForwardIterator it = this->get_bit_forward_iterator_begin(); it != _end; ++it)
512 {
513 uint64_t bits = *it;
514 uint64_t i = 0;
515 while (i < 64 && counter < _size)
516 {
517 bool b = stool::Byte::get_bit(bits, i);
518 r[counter] = b;
519 counter++;
520 i++;
521 }
522 }
523 assert(counter == _size);
