510 std::vector<bool> to_vector() const
511 {
512 uint64_t _size = this->size();
513 std::vector<bool> r;
514 r.resize(_size, false);
515 uint64_t counter = 0;
516 auto _end = this->get_leaf_forward_iterator_end();
517 for (BitForwardIterator it = this->get_bit_forward_iterator_begin(); it != _end; ++it)
518 {
519 uint64_t bits = *it;
520 uint64_t i = 0;
521 while (i < 64 && counter < _size)
522 {
523 bool b = stool::Byte::get_bit(bits, i);
524 r[counter] = b;
525 counter++;
526 i++;
527 }
528 }
529 assert(counter == _size);
