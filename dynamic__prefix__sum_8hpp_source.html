173 int64_t predecessor_index(uint64_t value) const
174 {
175 int64_t size = this->size();
176
177 if (size > 0)
178 {
179 uint64_t fst_value = this->at(0);
180 if (value >= fst_value)
181 {
182 if (value > this->psum())
183 {
184 return size - 1;
185 }
186 else
187 {
188 int64_t idx = this->search(value);
189 if (idx >= size)
190 {
191 return size - 1;
192 }
193 else
194 {
195 uint64_t v = this->psum(idx);
196
197 assert(v >= value);
198 if (v > value)
199 {
200 assert(idx - 1 < size);
201 return idx - 1;
202 }
