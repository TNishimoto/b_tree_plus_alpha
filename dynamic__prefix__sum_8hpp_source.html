271 {
272 int64_t size = this->size();
273
274 if (size > 0)
275 {
276 uint64_t fst_value = this->at(0);
277 if (x >= fst_value)
278 {
279 if (x > this->psum())
280 {
281 return size - 1;
282 }
283 else
284 {
285 int64_t idx = this->search(x);
286 if (idx >= size)
287 {
288 return size - 1;
289 }
290 else
291 {
292 uint64_t v = this->psum(idx);
293
294 assert(v >= x);
295 if (v > x)

297 assert(idx - 1 < size);
298 return idx - 1;
299 }
