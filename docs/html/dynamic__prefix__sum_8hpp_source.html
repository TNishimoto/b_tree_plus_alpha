178 {
179 int64_t size = this->size();
180
181 if (size > 0)
182 {
183 uint64_t fst_value = this->at(0);
184 if (value >= fst_value)
185 {
186 if (value > this->psum())
187 {
188 return size - 1;
189 }
190 else
191 {
192 int64_t idx = this->search(value);
193 if (idx >= size)
194 {
195 return size - 1;

201 assert(v >= value);
